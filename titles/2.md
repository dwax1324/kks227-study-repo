#### 3085번: 사탕 게임

나중에 다시 풀어보기

</summary>

#define fastio ios_base::sync_with_stdio(false), cin.tie(NULL), cout.tie(NULL)

#define all(x) (x).begin(), (x).end()

void misc(int32_t &t);
/* ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾ ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾ ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾

    comment: 

⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾ ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾ ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾*/


char arr[51][51];

int n;
int checkRowCol(int y, int x, char t){
    int cntRow=1;
    int cntCol=1;
    int ny=y,nx=x+1;
    while(1){
        if(nx < n && arr[ny][nx] == t){
            cntRow++;
        }else break;
        nx++;
    }
    ny =y, nx = x-1;
    while(1){
        if(nx>=0 && arr[ny][nx] == t){
            cntRow++;
        }else break;
        nx--;
    }
    ny= y+1, nx = x;
    while(1){
        if(ny<n && arr[ny][nx] == t){
                cntCol++;
        }else break;
        ny++;
    }
    ny = y-1, nx= x;
    while(1){
        if(ny>=0 && arr[ny][nx] == t){
            cntCol++;
        }
        else break;
        ny--;
    }
    return max(cntRow,cntCol);
}

int cntCandies(){
    int ret=0;
    for(int i=0; i < n; i++){
        for(int j=0; j <n ;j++){
            ret= max(ret,checkRowCol(i,j,arr[i][j]));
        }
    }
    return ret;
}

int ans=0;
int vi[51][51];
void go(int y, int x){
    if(y>= n || x>= n) return;
    if(vi[y][x]) return;
    vi[y][x] = true;
    if(x-1 >= 0){
        swap(arr[y][x], arr[y][x-1]);
        ans = max(ans,cntCandies());        
        swap(arr[y][x], arr[y][x-1]);
    }
    if(x+1 < n){
        swap(arr[y][x], arr[y][x+1]);
        ans = max(ans,cntCandies());        
        swap(arr[y][x], arr[y][x+1]);
    }
    if(y-1 >= 0){
        swap(arr[y][x], arr[y-1][x]);
        ans = max(ans,cntCandies());        
        swap(arr[y][x], arr[y-1][x]);
    }

    if(y+1 < n){
        swap(arr[y][x], arr[y+1][x]);
        ans = max(ans,cntCandies());        
        swap(arr[y][x], arr[y+1][x]);
    }
    // printf("y: %d, x : %d\n",y,x);
    go(y+1,x);
    go(y,x+1);
}


void solve(){
    cin >> n;
    
    for(int i=0; i < n; i++){
        for(int j=0; j< n; j++){
            cin >> arr[i][j];
        }
    }

0이 있는 수이거나 겹치는 숫자가 있는 수면 걸러주고  
입력마다 123~987까지 loop을 돌면서 조건에 만족하면 1을 증가시킨다.  
결국 n개의 조건을 모두 만족하는 숫자만이 영수가 생각하고 있을 가능성이 있는 답이다.
