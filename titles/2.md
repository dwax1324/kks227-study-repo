#### 3085번: 사탕 게임

함정이 많은 문제다.  

나중에 다시 풀어보기  

1.dfs로 풀면 안된다.  
2.행과 열 각각 체크해야한다.  
3.바꾼곳 뿐만 아니라 전체의 맵을 체크해야한다.  
</summary>

```cpp
#include <bits/stdc++.h>
using namespace std;
#define fastio ios_base::sync_with_stdio(false), cin.tie(NULL), cout.tie(NULL)
#define debug freopen("input.txt", "r", stdin), freopen("output.txt", "w", stdout)
#define sz(x) (int)(x).size()
#define all(x) (x).begin(), (x).end()
#define rall(x) (x).rbegin(), (x).rend()
// #define int int64_t
void misc(int32_t &t);
/* ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾ ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾ ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾
    2020.01.02 Sat
    comment: 

⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾ ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾ ⁽⁽◝( ˙ ꒳ ˙ )◜⁾⁾*/


char arr[51][51];

int n;
int checkRowCol(int y, int x, char t){
    int cntRow=1;
    int cntCol=1;
    int ny=y,nx=x+1;
    while(1){
        if(nx < n && arr[ny][nx] == t){
            cntRow++;
        }else break;
        nx++;
    }
    ny =y, nx = x-1;
    while(1){
        if(nx>=0 && arr[ny][nx] == t){
            cntRow++;
        }else break;
        nx--;
    }
    ny= y+1, nx = x;
    while(1){
        if(ny<n && arr[ny][nx] == t){
                cntCol++;
        }else break;
        ny++;
    }
    ny = y-1, nx= x;
    while(1){
        if(ny>=0 && arr[ny][nx] == t){
            cntCol++;
        }
        else break;
        ny--;
    }
    return max(cntRow,cntCol);
}

int cntCandies(){
    int ret=0;
    for(int i=0; i < n; i++){
        for(int j=0; j <n ;j++){
            ret= max(ret,checkRowCol(i,j,arr[i][j]));
        }
    }
    return ret;
}

int ans=0;
int vi[51][51];
void go(int y, int x){
    if(y>= n || x>= n) return;
    if(vi[y][x]) return;
    vi[y][x] = true;
    if(x-1 >= 0){
        swap(arr[y][x], arr[y][x-1]);
        ans = max(ans,cntCandies());        
        swap(arr[y][x], arr[y][x-1]);
    }
    if(x+1 < n){
        swap(arr[y][x], arr[y][x+1]);
        ans = max(ans,cntCandies());        
        swap(arr[y][x], arr[y][x+1]);
    }
    if(y-1 >= 0){
        swap(arr[y][x], arr[y-1][x]);
        ans = max(ans,cntCandies());        
        swap(arr[y][x], arr[y-1][x]);
    }

    if(y+1 < n){
        swap(arr[y][x], arr[y+1][x]);
        ans = max(ans,cntCandies());        
        swap(arr[y][x], arr[y+1][x]);
    }
    // printf("y: %d, x : %d\n",y,x);
    go(y+1,x);
    go(y,x+1);
}


void solve(){
    cin >> n;
    
    for(int i=0; i < n; i++){
        for(int j=0; j< n; j++){
            cin >> arr[i][j];
        }
    }
    
    go(0,0);
    cout << ans;

}


int32_t main() {
    int32_t T = 1;
    misc(T);
    // cin >> T;
    for (int i = 1; i <= T; i++) {
        // cout << "Data Set "<< i << ":\n";
        // cout << "case #" << i << ": ";
        solve();
    }
}

    fastio;
   debug;
```
